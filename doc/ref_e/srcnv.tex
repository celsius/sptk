\name{srcnv}{サンプリングレート変換（アップサンプリング）}%
{サンプリングレート変換}

\begin{synopsis}
\item [srcnv] [ --s $S$ ] [ --c $C$ ] [ --u $U$ ] [ --d $D$ ] [ {\em infile} ]
\end{synopsis}

\begin{qsection}{DESCRIPTION}
 {\em srcnv}アップサンプリングを行います．
\par
データ形式は入力，出力ともfloat形式です．
フィルタ係数は次のものが用いられます．

\begin{tabular}{ll} \\[-1ex]
	$S=23F$ & \$SPTK/lib/lpfcoef.2to3f \\
	$S=23S$ & \$SPTK/lib/lpfcoef.2to3s \\
	$S=34$ & \$SPTK/lib/lpfcoef.3to4 \\
	$S=45$ & \$SPTK/lib/lpfcoef.4to5 \\
	$S=57$ & \$SPTK/lib/lpfcoef.5to7 \\
	$S=58$ & \$SPTK/lib/lpfcoef.5to8 \\
        &(\$SPTKはインストールしたディレクトリ)
\end{tabular}

{\bf --u，--d}オプションで，アップサンプリングとダウンサンプリングの比
率を変更することができます．フィルタ係数のファイルを指定する場合は，
{\bf --u，--d}オプションも同時に指定します．

なお，ファイルタ係数のファイルはASCII形式となっています．
\end{qsection}

\begin{options}
	\argm{s}{S}{変換タイプ． \\
		\begin{tabular}{ll} \\[-1ex]
			$S=23F$ & 比率 $2 : 3$ でアップサンプリング \\
			$S=23S$ & 比率 $2 : 3$ でアップサンプリング \\
			$S=34$ & 比率 $3 : 4$ でアップサンプリング \\
			$S=45$ & 比率 $4 : 5$ でアップサンプリング \\
			$S=57$ & 比率 $5 : 7$ でアップサンプリング \\
			$S=58$ & 比率 $5 : 8$ でアップサンプリング
		\end{tabular}\\}{58}
	\argm{c}{\mbox{\em file}}{ローパスフィルタ係数ファイル名．}{Default}
	\argm{u}{U}{アップサンプリングの比率}{N/A}
	\argm{d}{D}{ダウンサンプリングの比率}{N/A}
\end{options}

\begin{qsection}{EXAMPLE}
float形式の標本化周波数10kHzの音声データファイル {\em data.10} を標本
化周波数16kHzにアップサンプリングし，{\em data.16} に出力する:
\begin{quote}
\verb! srcnv -s 58 data.10 > data.16!
\end{quote}
\end{qsection}
