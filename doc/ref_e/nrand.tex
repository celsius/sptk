\name{nrand}{generate normal distributed random value}{signal generation}

\begin{synopsis}
\item[nrand] [ --l $L$ ] [ --s $S$ ]
\end{synopsis}

\begin{qsection}{DESCRIPTION}
{\em nrand} は，正規雑音を発生し，標準出力に出力します．
内部ではnrandとsrndを使っています．
\par
Output data is in float format.
\end{qsection}

\begin{options}
	\argm{l}{L}{output length\\
			if $L \le 0$ then 無限に生成します．}{256}
	\argm{s}{S}{seed for nrand}{1}
\end{options}

\begin{qsection}{EXAMPLE}
長さ100の正規雑音系列を発生し {\em data.rnd} に出力する:
\begin{quote}
\verb!nrand -l 100 -s 3 > data.rnd!
\end{quote}
\end{qsection}
