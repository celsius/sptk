\name{bell}{ベルを鳴らす}{その他}

\begin{synopsis}
\item[bell] [ $num$ ] 
\end{synopsis}

\begin{qsection}{DESCRIPTION}
{\em bell}は，オプションで指定した回数だけベルを鳴らします．
\end{qsection}

\begin{options}
	\desc{\hspace{\tabcolsep}$num$
		\hspace{2\tabcolsep}ベルを鳴らす回数．
		\hfill [1]}
\end{options}

\begin{qsection}{EXAMPLE}
ベルを10回鳴らす :
\begin{quote}
 \verb!bell 10!
\end{quote}
\end{qsection}
