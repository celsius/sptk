\name[ref:gcep-IEICE,ref:gcep-IEEEASSP,ref:gcep-ICASSP90]{gcep}%
{一般化ケプストラム分析}{音声分析}

\begin{synopsis}
\item [gcep] [ --m $M$ ] [ --g $G$ ] [ --l $L$ ] [ --n ]
	     [ --i $I$ ] [ --j $J$ ] [ --d $D$ ] [ --e $E$ ]
\item [\ ~~~~~~~] [ {\em infile} ]
\end{synopsis}

\begin{qsection}{DESCRIPTION}
一般化ケプストラム分析を行い，
正規化一般化ケプストラム $c_\gamma'(m)$ を標準出力に出力します．
入力は窓掛けされた長さ $L$ の時系列
\begin{displaymath}
  x(0),x(1),\ldots,x(L-1)
\end{displaymath}
です．
\par
データ形式は入力，出力ともfloat 形式です．
\par
一般化ケプストラム分析では、音声スペクトルを $M$ 次の一般化ケプストラム 
$c_\gamma(m)$ または正規化一般化ケプストラム $c_\gamma'(m)$ により
\begin{eqnarray*}
H(z) &=& s_\gamma^{-1}\left(
	\sum_{m=0}^{M} c_\gamma(m)z^{-m} \right) \\
     &=& K \cdot s_\gamma^{-1}\left(
	\sum_{m=1}^{M} c_\gamma'(m)z^{-m} \right) \\
     &=& \left\{ \begin{array}{ll} \displaystyle
	K\cdot \left( 1+\gamma\sum_{m=1}^{M} c_\gamma'(m)z^{-m}
		\right)^{1/\gamma}, & -1 \leq \gamma < 0 \\
	\displaystyle K\cdot \exp \sum_{m=1}^{M} c_\gamma'(m)z^{-m}, 
		& \gamma=0
	\end{array} \right.
\end{eqnarray*}
とモデル化し，対数スペクトルの不偏推定法 
における評価関数を適用します．
評価関数の最小化は、 $\gamma=-1$ のときには線形予測法での
線形方程式を解くことと等価になりますが，
$\gamma=-1$ 以外では非線形方程式を解くことになりますので，
ここではNewton--Raphson法を利用して解を求めています．
\end{qsection}

\begin{options}
	\argm{m}{M}{分析次数．}{25}
	\argm{g}{G}{一般化ケプストラムのべきパラメータ $\gamma$．\\
			 ただし，$G>1.0$ のときは $\gamma=-1/G$．}{0}
	\argm{l}{L}{入力データのフレーム長．}{256}
	\argm{n}{}{正規化したケプストラムを出力．}{FALSE}
	\desc[1zh]{通常，以下のオプションの指定は必要ありません．}
	\argm{i}{I}{分析の最小反復回数．}{2}
	\argm{j}{J}{分析の最大反復回数．}{30}
	\argm{d}{D}{Newton-Raphson 法の終了条件．デフォルトは $D=0.001$ で，
		    このときは $\varepsilon^{(i)}$ の繰り返しによる変化率が
		    $0.001$ つまり $0.1\%$ 以内になったときに終了．}{0.001}
	\argm{e}{E}{ピリオドグラムに足し込む小さな値．}{0}
\end{options}

\begin{qsection}{EXAMPLE}
float 形式のファイル {\em data.f} を次数15次で一般化ケプストラム分析し，
一般化ケプストラムを {\em data.gcep} に出力する:
\begin{quote}
 \verb!frame < data.f | window | gcep -m 15 > data.gcep!
\end{quote} 
\end{qsection}
