\name{echo2}{output of the standard error}{others}

\begin{synopsis}
\item[echo2] [ --n ] [ argument ]
\end{synopsis}

\begin{qsection}{DESCRIPTION}

The command {\em echo2} outputs its arguments to the standard error
output.

\end{qsection}

\begin{options}
	\argm{n}{}{no output newline}{}
\end{options}

\begin{qsection}{EXAMPLE}
This example prints ''error!'' in the standard error output:
\begin{quote}
  \begin{verbatim}
  echo2 -n "error!"
  \end{verbatim}
\end{quote}
\end{qsection}
