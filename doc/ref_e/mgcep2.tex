\name[ref:mgcep-IEICE,ref:mgcep-ICSLP94]{mgcep2}%
{高速メル一般化ケプストラム分析($\gamma = -\frac{1}{2}$)}%
{音声分析}

\begin{synopsis}
\item[mgcep2]   [ --a $A$ ] [ --m $M$ ] [ --l $L$ ] 
	       [ --o $O$ ] [ --i $I$ ] [ --j $J$ ] [ --d $D$ ]
		 [ {\em infile} ]
\end{synopsis}

\begin{qsection}{DESCRIPTION}
メル一般化ケプストラム分析を行います．
$\gamma = -\frac{1}{2}$とすることで高速化しています．
分析結果は，{\bf --o} オプションの指定に従った形式で，
標準出力に出力します．
入力は窓掛けされた長さ $L$ の時系列
\begin{displaymath}
  x(0),x(1),\ldots,x(L-1)
\end{displaymath}
です．
\par
データ形式は入力，出力ともfloat 形式です．
\par
メル一般化ケプストラム分析では，音声のスペクトルを $M$ 次の
メル一般化ケプストラム $c_{\alpha,\gamma}(m),\gamma = -\frac{1}{2}$ により
\begin{eqnarray*}
H(z) &=& s_{-\frac{1}{2}}^{-1}\left(
	\sum_{m=0}^M c_{\alpha, -\frac{1}{2}}(m)z^{-m} \right) \\
     &=& \displaystyle
	\left( 1-\frac{1}{2}\sum_{m=1}^M c_{\alpha, -\frac{1}{2}}(m)\tilde{z}^{-m}
		\right)^{-2}
\end{eqnarray*}
とモデル化し，対数スペクトルの不偏推定法
における評価関数を適用します．
ここで， $\tilde{z}^{-1}$ は1次のオールパス関数
\begin{displaymath}
\tilde{z}^{-1} = \frac{z^{-1}-\alpha}{1-\alpha z^{-1}}
\end{displaymath}
であり，その位相特性は標本化周波数10kHzの場合 $\alpha=0.35$ ，
標本化周波数8kHzの場合 $\alpha=0.31$ と選べば，
人間の音の高さに対する聴覚特性を表すメル尺度をよく近似します．
\par
ここでは，評価関数を最小にするメル一般化ケプストラムを求めるために，
Newton--Raphson法が用いられています．
\par

\newpage
\begin{options}
	\argm{a}{A}{周波数圧縮パラメータ $\alpha$ ．}{0.35}
	\argm{m}{M}{分析次数．}{25}
	\argm{l}{L}{FFT の点数．2 のべき乗数で指定．}{256}
	\argm{o}{O}{$O = 0$ のとき
			\begin{displaymath}
			  c_{\alpha, \gamma}(0), c_{\alpha, \gamma}(1), 
			  \ldots, c_{\alpha, \gamma}(M)
			\end{displaymath}
			$O = 1$ のとき
			\begin{displaymath}
			  b_\gamma(0), b_\gamma(1), \ldots, b_\gamma(M)
			\end{displaymath}
			$O = 2$ のとき
			\begin{displaymath}
			  K_\alpha, c_{\alpha, \gamma}'(1), 
			  \ldots, c_{\alpha, \gamma}'(M)
			\end{displaymath}
			$O = 3$ のとき
			\begin{displaymath}
			  K, b_\gamma'(1), \ldots, b_\gamma'(M)
			\end{displaymath}
			$O = 4$ のとき
			\begin{displaymath}
			  K_\alpha, \gamma\,c_{\alpha, \gamma}'(1), \ldots,
			\gamma\,c_{\alpha, \gamma}'(M)
			\end{displaymath}
			$O = 5$ のとき
			\begin{displaymath}
			  K, \gamma\,b_\gamma'(1), \ldots, 
			  \gamma\,b_\gamma'(M)
			\end{displaymath}
			が標準出力に出力されます．}{0}
	\desc[1ex]{通常，以下のオプションの指定は必要ありません．}
	\argm{i}{I}{Newton-Raphson法の最小反復回数．}{2}
	\argm{j}{J}{Newton-Raphson法の最大反復回数．}{30}
	\argm{d}{D}{Newton-Raphson 法の終了条件．
			デフォルトは$D=0.001$ で，
			このとき$\varepsilon^{(i)}$ の繰り返しによる変化率が
			$0.001$ つまり$0.1\%$ 以内になったとき終了します．
			}{0.001}
\end{options}
\end{qsection}

\begin{qsection}{EXAMPLE}
float 形式の音声データ{\em data.f} を$\gamma=-\frac{1}{2}$, $\alpha=0$
 (つまり対数スペクトルの不偏推定法)で分析し，{\em data.cep} に
ケプストラム係数を得る:
\begin{quote}
  \verb!frame < data.f | window | mgcep2 > data.cep !
\end{quote}
\end{qsection}
