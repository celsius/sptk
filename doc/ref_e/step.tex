\name{step}{generate step sequence}{signal generation}

\begin{synopsis}
\item[step] [ --l $L$ ] [ --n $N$ ]
\end{synopsis}

\begin{qsection}{DESCRIPTION}
長さ$L$ または次数$N$のユニットステップを標準出力に出力します．つまり，
\begin{displaymath}
\underbrace{1, 1, 1, \ldots, 1}_{L}
\end{displaymath}
\par
Output data is in float format.
\end{qsection}

\begin{options}
	\argm{l}{L}{length\\
                    $ L \le 0 $の場合，無限にステップ列を生成．}{256}
	\argm{n}{N}{order}{255}
\end{options}

\begin{qsection}{EXAMPLE}
標準形のディジタルフィルタにユニットステップ列を加える:
\begin{quote}
\verb!step | dfs -a 1 -0.8 | dmp!
\end{quote}
\end{qsection}
