\name{cdist}{ケプストラム距離の計算}{データ処理}

\begin{synopsis}
\item [cdist] [ --m $M$ ] [ --o $O$ ] [ --f ] {\em cfile}
 	    [ {\em infile} ] 
\end{synopsis}

\begin{qsection}{DESCRIPTION}
ファイル {\em infile}（省略時は標準入力）とファイル {\em cfile}との
ケプストラム距離を計算します．
{\em infile}のフレーム$t$のケプストラムを
\[c_{1,t}(0),c_{1,t}(1),c_{1,t}(2),\cdots,c_{1,t}(M)\]
{\em cfile}のフレーム$t$のケプストラムを
\[c_{2,t}(0),c_{2,t}(1),c_{2,t}(2),\cdots,c_{2,t}(M)\]
とした時，フレームごとの自乗ケプストラム距離は
\[d(t)=\sum_{k=1}^{M} (c_{1,t}(k)-c_{2,t}(k))^2\]
となり，ファイル全体のケプストラム距離は
\[d=\frac{1}{T} \sum_{t=0}^{T} d(t)\]
となります．

但し、$T$ は {\em infile}，{\em cfile} のうち少ない方のフレーム数
となります．


\end{qsection}

\begin{options}
	\argm{m}{M}{最小位相ケプストラムの次数．}{25}
	\argm{o}{O}{ケプストラム距離のフォーマット．\\
		\begin{tabular}{ll} \\[-1zh]
		$O=0$ & $  \frac{10}{\ln 10} \sqrt{2d(t)}$ \ \ \ \ [db]\\
		$O=1$ & $  d(t)$ \\
		$O=2$ & $  \sqrt{d(t)}$ \\[1zh]
		\end{tabular}\\
						}{0}
	
	\argm{f}{}{フレームごとのケプストラム距離を出力}{FALSE}
\end{options}

\begin{qsection}{EXAMPLE}
float 形式の15次のケプストラムファイル {\em data1.cep}，{\em data2.cep}
との自乗ケプストラム距離を求め，表示します．
\begin{quote}
\verb! cdist -m 15 data1.cep data2.cep | dmp !
\end{quote}
\end{qsection}
