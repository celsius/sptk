\name{interpolate}{データ列の補間}{信号処理}

\begin{synopsis}
\item[interpolate] [ --p $P$ ] [ --s $S$ ] [ {\em infile} ]
\end{synopsis}

\begin{qsection}{DESCRIPTION}
標準入力から入力される時系列
\begin{displaymath}
x(0), x(1), x(2), \ldots, 
\end{displaymath}
から周期 $P$ で
\begin{displaymath}
\underbrace{0, 0, \ldots, 0}_{S-1},\underbrace{x(0), 0, 0, \ldots, 0}_{P},\underbrace{x(1), 0, 0, \ldots, 0}_{P},x(2), \ldots, 
\end{displaymath}
のように0を補間します．
\par
データ形式は入力，出力ともfloat 形式です．
\end{qsection}

\begin{options}
	\argm{p}{P}{補間周期．}{10}
	\argm{s}{S}{開始サンプル．}{0}
\end{options}

\begin{qsection}{EXAMPLE}
float形式のデータ {\em data.f} を周期2で decimation し，さらに0を周期2で
interpolation して {\em data.di} に出力する:
\begin{quote}
  \verb!decimate -p 2  < data.f | interpolate -p 2 > data.di!
\end{quote}
\end{qsection}
