\name{mgc2sp}{メル一般化ケプストラムから対数振幅スペクトルを求める}%
{音声パラメータ変換}

\begin{synopsis}
\item[mgc2sp] [ --a $A$ ] [ --g $G$ ] [ --m $M$ ]
	       [ --n ] [ --u ] [ --l $L$ ] [ --p ]
\item[\ ~~~~~] [ --o $O$ ] [ {\em infile} ]
\end{synopsis}

\begin{qsection}{DESCRIPTION}
入力されたメル一般化ケプストラム $c_{\alpha, \gamma}(m)$ から
対数振幅スペクトルを求めます．
\par
データ形式は入力，出力ともfloat 形式です．
\par
メル一般化ケプストラム $c_{\alpha, \gamma}(m)$ をメル一般化対数変換
（mgc2mgc 参照）によりケプストラムに変換し，ケプストラムから対数振幅
スペクトルを求めます（spec 参照）．

ゲインで正規化されたものを入力する場合は、入力が
\begin{eqnarray*}
\hspace{-15mm}&&K_{\alpha} = 
	s_{\gamma}^{-1}\left(c_{\alpha,\gamma}^{(0)}(0)\right), 
	  \qquad\qquad \\ 
\hspace{-15mm}&&c_{\alpha,\gamma}'(m) =
          c_{\alpha,\gamma}^{(0)}(m)/\left(1+\gamma\,
	  c_{\alpha,\gamma}^{(0)}(0)\right), \quad m = 1,2,\ldots, M 
\end{eqnarray*}
の形であるとみなします．

また，$\gamma$を掛けた形で入力する場合は，\\
入力がゲインで正規化されていない場合，
\begin{displaymath}
1+\gamma c_{\alpha,\gamma}(0), \gamma c_{\alpha,\gamma}(1), \ldots, \gamma c_{\alpha,\gamma}(M)
\end{displaymath}
正規化されている場合，
\begin{displaymath}
K_\alpha,\gamma c_{\alpha,\gamma}'(1),\ldots, \gamma c_{\alpha,\gamma}'(M)
\end{displaymath}
の形であるとみなします．

\end{qsection}

\begin{options}
	\argm{a}{A}{周波数圧縮パラメータ $\alpha$．}{0}
	\argm{g}{G}{一般化ケプストラムのべきパラメータ $\gamma$ ．\\
			 ただし，$G>1.0$ のときは $\gamma = -1/G$．}{0}
	\argm{m}{M}{入力されるメル一般化ケプストラムの次数．}{25}
	\argm{n}{}{入力を正規化されたメル一般化ケプストラムとみなす．}{FALSE}
	\argm{u}{}{入力を$\gamma$を掛けた形とみなす．}{FALSE}
	\argm{l}{L}{FFT長．}{256}
	\argm{p}{}{スペクトルの位相を出力．}{FALSE}
	\argm{o}{O}{--pオプションが指定されていない場合,出力するスペクトルの
		    スケールを指定． \\
		\begin{tabular}{ll} \\[-1zh]
			$O=0$ & $20 \times \log |H(z)|$ \\
			$O=1$ & $\ln |H(z)|$ \\
			$O=2$ & $|H(z)|$ \\[1zh]
		\end{tabular} \\
		 --pオプションが指定されている場合,出力する位相の単位を指定．
		\\
		\begin{tabular}{ll} \\[-1zh]
			$O=0$ & $\arg |H(z)| \div \pi \quad [\pi \; rad.]$ \\
			$O=1$ & $\arg |H(z)| \quad [rad.]$ \\
			$O=2$ & $\arg |H(z)| \times180\div\pi\quad[deg.]$ \\
		\end{tabular}\\\hspace*{\fill}}{0}

\end{options}

\begin{qsection}{EXAMPLE}
float形式のメル一般化ケプストラムファイル {\em data.mgcep} 
($M=12, \alpha=0.35, \gamma=-0.5$)から対数振幅スペクトルを求め，表示する:
\begin{quote}
 \verb!mgc2sp -m 12 -a 0.35 -r -0.5 < data.mgcep | glogsp |gr!
\end{quote} 
\end{qsection}
