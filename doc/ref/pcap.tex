\name{pcap}{主成分得点を計算する}{信号処理}
\def\Vec#1{\mbox{\boldmath $#1$}}

\begin{synopsis}
\item[pca] [ --l $L$ ] [ --n $N$] {\em pcafile} [ {\em infile} ] 
\end{synopsis}

\begin{qsection}{DESCRIPTION}
 主成分分析によって得られた固有ベクトルおよび
 分布の平均ベクトルを {\em pcafile} から読み込み, 主成分得点を計算します．
 なお平均ベクトルのデータは{\em pcafile}の最後尾に置かれます．
 データ形式は入出力ともfloat 形です．
\end{qsection}

\begin{options}
 \argm{l}{L}{ベクトルの次元数．}{3}
 \argm{n}{N}{分析次数．}{2}
\end{options}

\begin{qsection}{EXAMPLE}
 3次元の固有ベクトルおよび分布の平均ベクトルが格納された
 float形式のファイル {\em pca.dat} を用い，
 同様の$3$次元ベクトルの集合からなる
 テストデータ {\em test.dat} の主成分得点を
 分析次数2次で計算し {\em score.dat} へ出力する:
 \begin{quote}
  \verb!pcap pca.dat -l 3 -n 2 < test.dat > score.dat!
 \end{quote}
\end{qsection}
