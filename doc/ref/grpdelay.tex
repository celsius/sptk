\name{grpdelay}{郡遅延を求める}
\begin{synopsis}
 \item[grpdelay] [ --l $L$ ] [ --m $M$ ] [ --a ] [ {\em infile} ] 
\end{synopsis}

\begin{qsection}{DESCRIPTION}
ファイル{\em infile}(省略時は標準入力)からフィルタ係数(FIRフィルタ)
を読み込み，郡遅延を出力します．データ形式は入出力は共にfloat形式です．

{\bf --m} オプションで $M$ の指定がなく，かつ入力系列長がFFT サイズより短い
場合，入力系列に0 を付加してFFT を実行します．
{\bf --a}オプションが指定された場合，入力の零次の項はゲイン項として取り扱い
ます．
\par
\[
\begin{array}{lll}
\mbox{入力系列} & 
\overbrace{\framebox[4.5cm]{$x_0, x_1, \ldots, x_{M}, 0,
					\ldots,0$}}^{L}  & \mbox{フィルタ係数}\\
		& \makebox[4.5cm]{0\hfill $L-1$} &
\end{array}
\]
\[
\begin{array}{lll}
\mbox{出力系列} & \overbrace{\framebox[4.5cm]{$\tau(\omega)$}}^{L/2+1} &
	   \mbox{郡遅延}\\
		& \makebox[4.5cm]{0\hfill $L-1$} &

\end{array}
\]
\end{qsection}

\begin{options}
	\argm{l}{L}{フーリエ変換のサイズ．2のべき乗で指定．}{256}
	\argm{m}{M}{フィルタの次数．}{L-1}
	\argm{a}{}{入力系列をARフィルタの係数とみなす．}{FALSE}
\end{options}


\begin{qsection}{EXAMPLE}
伝達関数
\begin{displaymath}
  H(z)=\frac{1}{1+0.9z^{-1}}
\end{displaymath}
をもつフィルタのインパルス応答の群遅延を求め，画面にプロットさせる:
\begin{quote}
\verb! impluse | dfs -a 1 0.9 | grpdelay | fdrw | xgr !
\end{quote}  
\end{qsection}
