\name{fftr}{実数列の高速フーリエ変換}{信号処理}

\begin{synopsis}
 \item[fftr] [ --l $L$ ] [ --m $M$] [ --\{ A $|$ R $|$ I $|$ P \} ] [ --H ]
	     [ {\em infile} ] 
\end{synopsis}

\begin{qsection}{DESCRIPTION}
実数列を{\em infile} から読み込み，FFT アルゴリズムによりDFT を実行します．
{\em infile} の指定がないときには，標準入力からデータが読み込まれます．
データ形式は入出力ともfloat 形です．
\par
{\bf --m} オプションで $M$ の指定がなく，かつ入力系列長がFFT サイズより
短い場合，入力系列に0 を付加してFFT を実行します．

\begin{displaymath}
\begin{array}{lll}
\mbox{入力系列} & 
\overbrace{\framebox[4.5cm]{$x_0, x_1, \ldots, x_{M}, 0,
					\ldots,0$}}^{L}  & \\
		& \makebox[4.5cm]{0\hfill $L-1$} &
\end{array}
\end{displaymath}
\begin{displaymath}
\begin{array}{lll}
\mbox{出力系列} & \overbrace{\framebox[4.5cm]{実　部}}^{L} &
	   \overbrace{\framebox[4.5cm]{虚　部}}^{L} \\
		& \makebox[4.5cm]{0\hfill $L-1$} &
		\makebox[4.5cm]{0\hfill $L-1$}
\end{array}
\end{displaymath}
\end{qsection}

\begin{options}
	\argm{l}{L}{フーリエ変換のサイズ．2のべき乗で指定．}{256}
	\argm{m}{M}{実系列の次数．}{L-1}
	\argm{A}{}{振幅スペクトルを出力．}{FALSE}
	\argm{R}{}{実部のみを出力．}{FALSE}
	\argm{I}{}{虚部のみを出力．}{FALSE}
	\argm{P}{}{パワースペクトルを出力．}{FALSE}
	\argm{H}{}{出力系列を $(L/2+1)$ 個とする．}{FALSE}
%	 \footnote{実部は偶関数，虚部は奇関数でもあることに注意．}
\end{options}

\begin{qsection}{EXAMPLE}
正弦波にブラックマン窓をかけ，DFT 行った結果の振幅をプロットする:
\begin{quote}
  \verb!sin -t 30 | window | fftr -A | fdrw | xgr!
\end{quote}

\end{qsection}
