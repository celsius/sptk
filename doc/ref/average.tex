\name{average}{ブロック毎の平均を計算する}{none}

\begin{synopsis}
\item [average] [ --l $L$ ] [ --n $N$ ] [ {\em infile} ] 
\end{synopsis}

\begin{qsection}{DESCRIPTION}
指定されたファイルから長さ $L$ の時系列
\begin{displaymath}
  x(0),x(1),\ldots,x(L-1)
\end{displaymath}
を読み込み,その平均
\begin{displaymath}
\frac{x(0)+x(1)+\ldots+x(L-1)}{L}
\end{displaymath}
を出力します．
ただし，$L=0$ のときにはファイル全体の平均を出力します．
\par
データ形式は入力，出力ともfloat 形式です．
\end{qsection}

\begin{options}
	\argm{l}{L}{平均をとるブロック長．}{0}
	\argm{n}{N}{平均をとるブロックの次数．}{L-1}
\end{options}

\begin{qsection}{EXAMPLE}
float形式のファイル {\em data.f} の平均を計算し，
{\em data.av} に出力する:
\begin{quote}
 \verb!average < data.f > data.av!
\end{quote} 
\end{qsection}
