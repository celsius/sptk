\name{norm0}{フィルタ係数を0次項で正規化する．}{信号処理}

\begin{synopsis}
\item[norm0] [ --m $M$ ] [ {\em infile} ]
\end{synopsis}

\begin{qsection}{DESCRIPTION}
標準入力から入力されるデータ列
\begin{displaymath}
x(0), x(1), \ldots, x(M)
\end{displaymath}
を0次の項で正規化
\begin{displaymath}
1/x(0), x(1)/x(0), \ldots, x(M)/x(0)
\end{displaymath}
を行い，出力します．
\par
データ形式は入力，出力とも float 形式です．
\end{qsection}

\begin{options}
        \argm{m}{M}{入力データの次数．}{25}
\end{options}
