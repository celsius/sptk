\name{snr}{SNRとセグメンタルSNRを求める}

\begin{synopsis}
\item [snr] [ --l $L$ ] [ +$t_1 t_2$ ] [ --o $O$ ] {\em file1} [ {\em infile} ] 
\end{synopsis}

\begin{qsection}{DESCRIPTION}
指定された2つのファイルからの入力
\begin{displaymath}
  x_1(0),x_1(1),\ldots,x_1(L-1)
\end{displaymath}
と
\begin{displaymath}
  x_2(0),x_2(1),\ldots,x_2(L-1)
\end{displaymath}
からSNR(Signal Noise Ratio)と
$\mbox{SNR}_{\mbox{seg}}$(セグメンタルSNR)を計算します．
\par
SNR， $\mbox{SNR}_{\mbox{seg}}$は次式で計算することができます．
\begin{displaymath}
\mbox{SNR} = 10~\log \frac{\displaystyle\sum_{n} ( x(n) )^{2}}
{\displaystyle\sum_{n} (e(n))^{2}}~~~\mbox{[dB]}
\end{displaymath}
\begin{displaymath}
\mbox{SNR}_{\mbox{seg}} = \frac{1}{N_{i}} \sum_{i = 1}^{N_{i}}
\mbox{SNR}_{i} ~~~\mbox{[dB]}
\end{displaymath}
ただし，
\begin{displaymath}
e(n) = x_1(n) - x_2(n)
\end{displaymath}
ここで， $N_i$はフレーム数を表しています．
セグメンタルSNRは子音区間のような
比較的小振幅区間のSNRも反映されるため，
主観値との対応が比較的良いという特徴があります．
\end{qsection}

\newpage
\begin{options}
        \argm{l}{L}{セグメンタルSNRを計算するときのフレーム長}{256}
        \argp{t_1 t_2}{入力データの形式．$t_1$，$t_2$はそれぞれ {\em file1}，
                {\em infile} のデータ型を表す． \\
                \begin{tabular}{llcll} \\[-1zh]
                        s & short型 (2bytes) & \quad &
                        f & float型 (4bytes) \\
                \end{tabular}\\\hspace*{\fill}}{sf}
        \argm{o}{O}{出力形式．\\
                        \begin{tabular}{ll} \\[-1zh]
                          0 & SNR and SNRseg\\
                          1 & SNR and SNRseg in detail\\
                          2 & SNR\\
                          3 & SNRseg
                        \end{tabular}\\\hspace*{\fill}\\
        0,1 はASCII形式で出力し 2,3 はfloat型で出力します}{0}
\end{options}

\begin{qsection}{EXAMPLE}
short形式のファイル {\em data.s} と float形式のファイル {\em data.f} のSNRと
セグメンタルSNRを出力する:
\begin{quote}
 \verb!snr +sf data.s data.f!
\end{quote} 
\end{qsection}
