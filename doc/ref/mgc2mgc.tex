\name{mgc2mgc}{メル一般化対数変換}%
{音声パラメータ変換}

\begin{synopsis}
 \item [mgc2mgc] [ --m $M_1$ ] [ --a $A_1$ ] [ --g $G_1$ ] [ --n ] [ --u ]
                 [ --M $M_2$ ] [ --A $A_2$ ] 
 \item [\ ~~~~~~~~~~~] [ --G $G_2$ ] [ --N ] [ --U ] [ {\em infile} ] 
\end{synopsis}

\begin{qsection}{DESCRIPTION}
入力されたメル一般化ケプストラム
$c_{\alpha_1,\gamma_1}(0),\ldots,c_{\alpha_1,\gamma_1}(M_1)$
に対してメル一般化対数変換を行い，メル一般化ケプストラム
$c_{\alpha_2,\gamma_2}(0),\ldots,c_{\alpha_2,\gamma_2}(M_2)$
を出力します．
\par
データ形式は入力，出力ともfloat 形式です．
\par
まず入力されたメル一般化ケプストラム $c_{\alpha_1,\gamma_1}(m)$ を
周波数変換( $\alpha_1 \rightarrow \alpha_2$ )を行い，
$c_{\alpha_2,\gamma_1}(m)$ を求めます．
\begin{eqnarray*} 
\hspace{-15mm}&& \alpha = (\alpha_2-\alpha_1)/(1-\alpha_1\alpha_2) \\
\hspace{-15mm}&& c_{\alpha_2,\gamma_1}^{(i)}(m) = \left\{ 
	\begin{array}{ll}
          c_{\alpha_1,\gamma_1}(-i)
	    +\alpha\,c_{\alpha_2,\gamma_1}^{(i-1)}(0), &  m=0 \\
          (1-\alpha^2)\,c_{\alpha_2,\gamma_1}^{(i-1)}(0)
            +\alpha\,c_{\alpha_2,\gamma_1}^{(i-1)}(1), &  m=1 \\
          c_{\alpha_2,\gamma_1}^{(i-1)}(m-1) 
	    +\alpha\, \left(c_{\alpha_2,\gamma_1}^{(i-1)}(m)
	    -c_{\alpha_2,\gamma_1}^{(i)}(m-1)\right), &   m=2,\ldots,M_2
         \end{array} \right\}, \quad \nonumber \\
\hspace{-15mm}&& \hspace{80mm} i = -M_1,\ldots,-1,0 
\end{eqnarray*}

\par
ゲインの正規化をして，$c_{\alpha_2,\gamma_1}'(m)$ を求めます．
\begin{eqnarray*}
\hspace{-15mm}&&K_{\alpha_2} = 
	s_{\gamma_1}^{-1}\left(c_{\alpha_2,\gamma_1}^{(0)}(0)\right), 
	  \qquad\qquad \\ 
\hspace{-15mm}&&c_{\alpha_2,\gamma_1}'(m) =
          c_{\alpha_2,\gamma_1}^{(0)}(m)/\left(1+\gamma_1\,
	  c_{\alpha_2,\gamma_1}^{(0)}(0)\right), \quad m = 1,2,\ldots, M_2 
\end{eqnarray*}

\par
$c_{\alpha_2,\gamma_1}'(m)$ に一般化対数変換( $\gamma_1 \rightarrow \gamma_2$ )
を行い，$c_{\alpha_2,\gamma_2}'(m)$ を求めます．
\begin{eqnarray*}
\hspace{-15mm}&&c_{\alpha_2,\gamma_2}'(m) =
        c_{\alpha_2,\gamma_1}'(m)+\sum_{k=1}^{m-1} \frac{k}{m}
          \left( \gamma_2\,c_{\alpha_2,\gamma_1}(k)\,
	  c_{\alpha_2,\gamma_2}'(m-k) \right. \nonumber \\
\hspace{-15mm}&& \hspace{30mm} \left. -\gamma_1\,c_{\alpha_2,\gamma_2}(k)\,
          c_{\alpha_2,\gamma_1}'(m-k) \right), 
	  \quad m = 1, 2, \ldots, M_2 
\end{eqnarray*}

\par
最後にゲインを逆正規化して，$c_{\alpha_2,\gamma_2}(m)$ を求めます．
\begin{eqnarray*}
\hspace{-15mm}&&c_{\alpha_2,\gamma_2}(0)= 
	s_{\gamma_2}\left(K_{\alpha_2}\right), \qquad \qquad \\
\hspace{-15mm}&&c_{\alpha_2,\gamma_2}(m) =
          c_{\alpha_2,\gamma_2}'(m)\,\left(1+\gamma_2\, 
          c_{\alpha_2,\gamma_2}(0)\right), 
          \quad m = 1,2,\ldots, M_2
\end{eqnarray*}

$\gamma$を掛けた形で入出力する場合は，
$c_{\alpha,\gamma}(m)$が正規化されていない場合は，
\begin{displaymath}
1+\gamma c_{\alpha,\gamma}(0), \gamma c_{\alpha,\gamma}(1), \ldots, \gamma c_{\alpha,\gamma}(M)
\end{displaymath}
とし，正規化されている場合は
\begin{displaymath}
K_\alpha,\gamma c_{\alpha,\gamma}'(1),\ldots, \gamma c_{\alpha,\gamma}'(M)
\end{displaymath}
とします．

\end{qsection}

\begin{options}
	\argm{m}{M_1}{入力されるメル一般化ケプストラムの次数．}{25}
	\argm{a}{A_1}{入力されるメル一般化ケプストラムの周波数圧縮
			   パラメータ$\alpha_1$．}{0}
	\argm{g}{G_1}{入力されるメル一般化ケプストラムのべきパラメータ
			$\gamma_1$．\\ ただし，$G_1 > 1.0$ のとき
			   は $\gamma_1 = -1 / G_1$．}{0}
	\argm{n}{}{入力をゲインが正規化されたメル一般化
					ケプストラムとみなす．}{FALSE}
	\argm{u}{}{入力を$\gamma_1$を掛けた形とみなす．}{FALSE}
	\argm{M}{M_2}{出力されるメル一般化ケプストラムの次数．}{25}
	\argm{A}{A_2}{出力されるメル一般化ケプストラムの周波数圧縮
			   パラメータ $\alpha_2$．}{0}
	\argm{G}{G_2}{出力されるメル一般化ケプストラムのべきパラメータ
			$\gamma_2$．\\ ただし，$G_2 > 1.0$ のとき
			   は $\gamma_2 =-1 / G_2$．}{1}
	\argm{N}{}{出力をゲインが正規化されたメル一般化
					ケプストラムとみなす．}{FALSE}
	\argm{U}{}{出力を$\gamma_2$を掛けた形とみなす．}{FALSE}
\end{options}

\begin{qsection}{EXAMPLE}
float形式の12次の線形予測係数ファイル {\em data.lpc} から，
30次のメルケプストラムを求め，{\em data.mcep} に出力する:
\begin{quote}
 \verb!mgc2mgc -m 12 -a 0 -g -1 -M 30 -A 0.31 -G 0!\\
 \verb!                     < data.lpc > data.mcep!
\end{quote} 
\end{qsection}
