\name{fd}{バイナリデータのダンプ}{データ操作}

\begin{synopsis}
 \item [fd] [ --a $A$ ] [ --n $N$ ] [ --m $M$ ] [ --{\em ent} ] 
	    [ +{\em type} ] [ $\%${\em form} ] [ {\em infile} ]
\end{synopsis}

\begin{qsection}{DESCRIPTION}
指定されたファイルの全データを，指定されたフォーマットでダンプします．
ファイルが指定されなかった場合，データは標準入力から読み込まれます．
\end{qsection}

\begin{options}
	\argm{a}{A}{アドレスつきで出力． $A$ は最初のアドレス．}{0}
	\argm{n}{N}{番号付きで出力． $N$ は最初の番号．}{0}
	\argm{m}{M}{番号付けを行なう場合に $M$ による modulo をとる．}{EOF}
	\argm{{\em ent}}{}{{\em ent}は数字で，1行に出力するデータの個数．}{0}
	\argp{t}{入力データの形式． \\
		\begin{tabular}{llcll} \\[-1zh]
			c & char型 (1byte) & \quad &
			s & short型 (2bytes) \\
			i & int型 (4bytes) & \quad &
			l & long型 (4bytes) \\
			f & float型 (4bytes) & \quad &
			d & double型 (8bytes) 
		\end{tabular}\\\hspace*{\fill}}{c}
        \argh{form}{}{出力のフォーマット指定．\\
			C言語のprintf関数の引数の形で出力のフォーマットを
			指定する．ただし，入力データの形式がf,d
                        の時のみ有効．}{N/A}
\end{options}

\begin{qsection}{EXAMPLE}
 音声データ sample.wav をアドレス付きでダンプする．
\begin{quote}
 \verb!fd -a 0 sample.wav!
\end{quote}
 結果\\
\verb!000000  52 49 46 46 9a 15 00 00 57 41 56 45 66 6d 74 20 |RIFF....WAVEfmt |!\\
\verb!000010  10 00 00 00 01 00 01 00 40 1f 00 00 40 1f 00 00 |........@...@...|!\\
\verb!000020  01 00 08 00 64 61 74 61 76 15 00 00 8a 8a 8f 99 |....datav.......|!

\end{qsection}
